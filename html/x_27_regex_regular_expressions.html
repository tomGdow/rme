Mathematica

StringSplit["ThomasDowling", RegularExpression["(?<=[a-z])(?=[A-Z])"]]
>{Thomas, Dowling}
                            

StringSplit["atgc", RegularExpression["(?<=[t])(?=[g])"]]
>{at, gc}                         
                            

StringCases["pre-12", RegularExpression["pre-\\K\\d{2}"]]
>{12}
                            

StringCases["GGGAAAZZZZZZDDD", RegularExpression["A(?R)?Z"]]
>{AAAZZZ}
