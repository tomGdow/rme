Examples
JS (Opening Example)
function drawOnCanvas() {
   let el = document.getElementsByTagName('canvas')[0] || null;
   if (el && el.getContext) {
   let ctx = el.getContext('2d');
   ctx.fillStyle = "red";
   ctx.fillRect(25, 50, 50, 50);
   ctx.clearRect(37.5, 62.5, 25, 25);
   ctx.strokeStyle = getComputedStyle(document.documentElement).getPropertyValue('--stroke-style')
   ctx.beginPath();
   ctx.arc(150, 75, 25, 0, 2 * Math.PI);
   ctx.stroke();
   ctx.closePath();
   ctx.beginPath();
   ctx.arc(150, 75, 12.5, 0, 2 * Math.PI);
   ctx.fill();
   ctx.closePath();
   ctx.beginPath();
   ctx.moveTo(250, 50);
   ctx.lineTo(275, 100);
   ctx.lineTo(225, 100);
   ctx.fill();
   ctx.closePath();
   }
}
Notes
default size
The default size of canvas is 300 (width) by 150 (height)
Preserve aspect ratio, otherwise distorts
(x,y)=(0,0)is at upper-right corner of canvas
Primitives
Two basic primitives: rectangles and path
