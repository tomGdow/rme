Examples
adjacent sibling selector 
div + div {
    border: solid lightcoral;
    width: 50%;
    margin: 10px;
    padding: 10px;
}
general sibling selector 
div ~ div {
    border: solid lightgreen;
    width: 50%;
    margin: 10px;
    padding: 10px;
}
grandchild selector 
div * div {
    border: solid orange;
    width: 50%;
    margin: 10px;
    padding: 10px;
}
great-grandchild selector 
div * *  div {
    border: solid lightgoldenrodyellow;
    width: 50%;
    margin: 10px;
    padding: 10px;
}
great-great-grandchild selector 
div * * * div {
    border: solid lightgoldenrodyellow;
    width: 50%;
    margin: 10px;
    padding: 10px;
}
